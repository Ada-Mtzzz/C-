#include <stdio.h>
int main()  
// inicia el programa
{
    float lado1, lado2, lado3, perimetro;
    // declaracion de variables
    printf("Cuanto mide el primer lado del triangulo \n");
    // mensaje para pedir el primer lado del triangulo
    scanf("%f", &lado1);
    // guarda el valor ingresado en lado1
   printf("Cuanto mide el segundo lado del triangulo \n");
    // mensaje para pedir el segundo lado del triangulo
    scanf("%f", &lado2);
    // guarda el valor ingresado en lado2
     printf("Cuanto mide el tercer lado del triangulo \n");
    // mensaje para pedir el tercer lado del triangulo
    scanf("%f", &lado3);
    // guarda el valor ingresado en lado3
    perimetro= lado1 + lado2 + lado3;
    // calcula el perimetro sumando los tres lados
    printf("El perimetro del triangulo es: %f\n", perimetro);
    // muestra el resultado del perimetro
    if (lado1 == lado2 && lado1 == lado3)
    // verifica si los tres lados son iguales
    {
        printf("El triangulo es equilatero");
        // si son iguales, es equilatero
    }
    if ((lado1 == lado2 && lado1 != lado3) || (lado1 == lado3 && lado1 != lado2) || (lado2 == lado3 && lado1 != lado2))
    // verifica si solo dos lados son iguales
    {
        printf("El triangulo es isosceles");
        // si solo dos son iguales, es isosceles
    }
    if (lado1 != lado2 && lado1 != lado3)
    // verifica si todos los lados son diferentes
    {
        printf("El triangulo es escaleno");
    // si todos son distintos, es escaleno
    }
        
}
